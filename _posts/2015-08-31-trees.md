```javascript
//Psuedo-classical instantiation
function Tree (value) {
  this.value = value;
  this.children = [];
};

//Prototypal instantiation
//This way lets us skip using 'new'
//but is generally less efficient.
var Tree = function (value) {
  var obj = Object.create(Tree.prototype);
  obj.value = value;
  obj.children = [];
  return obj;
};
```
